### **1. Настройка**

```bash
git config --global user.name "Имя"
git config --global user.email "email@example.com"
git config --global core.editor "code --wait"  # редактор по умолчанию
git config --list                              # список настроек
```

---

### **2. Создание и клонирование репозитория**

```bash
git init <project>        # создать новый репозиторий
git clone <url>           # склонировать репозиторий
```

---

### **3. Работа с файлами**

```bash
git status                # статус файлов
git add <file>            # добавить файл к коммиту
git add .                 # добавить все изменения
git rm <file>             # удалить файл из индекса и диска
git rm --cached <file>    # удалить только из индекса
git mv <old> <new>        # переименовать файл
```

---

### **4. Коммиты**

```bash
git commit -m "Сообщение"      # сделать коммит
git commit -am "Сообщение"     # добавить и закоммитить измененные файлы
git commit --amend             # исправить последний коммит
```

---

### **5. Ветвление**

```bash
git branch                   # список веток
git branch <name>            # создать ветку
git checkout <name>          # переключиться на ветку
git switch <name>            # новая форма checkout
git checkout -b <name>       # создать + перейти на ветку
git merge <branch>           # влить ветку в текущую
git branch -d <name>         # удалить ветку локально
```

---

### **6. Работа с удалёнными репозиториями**

```bash
git remote -v                # посмотреть удалённые
git remote add origin <url>  # добавить удалённый
git fetch                    # скачать изменения без слияния
git pull                     # скачать + смержить
git push                     # отправить коммиты
git push -u origin <branch>  # отправить и установить upstream
git push --force             # форсированная отправка
```

---

### **7. Просмотр истории**

```bash
git log                      # история коммитов
git log --oneline            # краткая история
git log --graph --all --oneline # граф веток
git diff                     # изменения по файлам
git diff <commit>            # изменения относительно коммита
git show <commit>            # детали конкретного коммита
```

---

### **8. Отмена изменений**

```bash
git restore <file>          # отмена изменений в файле
git restore --staged <file> # убрать из индекса
git reset --hard <commit>   # откат всех изменений к коммиту
git revert <commit>         # создать обратный коммит
```

---

### **9. Работа с тегами**

```bash
git tag                     # список тегов
git tag <name>               # создать тег
git push origin <name>       # отправить тег
```
